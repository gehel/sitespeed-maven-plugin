14  package ch.ledcom.maven.sitespeed;

16  import java.io.BufferedReader;
17  import java.io.File;
18  import java.io.IOException;
19  import java.io.InputStreamReader;
20  import java.net.URL;
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  import org.apache.maven.plugin.AbstractMojo;
25  import org.apache.maven.plugin.MojoExecutionException;
26  import org.apache.maven.plugin.MojoFailureException;
27  import org.apache.maven.plugins.annotations.Mojo;
28  import org.apache.maven.plugins.annotations.Parameter;
29  
30  import com.google.common.io.Closeables;
31  
32  /**
33   * Generate a SiteSpeed.io report.
34   * 
35   * At the moment this Mojo is a simple wrapper around the sitespeed.io bash
36   * script.
37   * 
38   * @author gehel
39   */
40  @Mojo(name = "sitespeed")
41  public class SiteSpeedMojo extends AbstractMojo {
42  
43      /** Path to the SiteSpeed.io directory. */
44      @Parameter(property = "siteSpeed.siteSpeePath", required = true)
45      private File siteSpeedPath;
46  
47      /** Path to the PhantomJS directory. */
48      @Parameter(property = "siteSpeed.phantomJSPath", required = true)
49      private File phantomJSPath;
50  
51      /** The start url for the test. */
52      @Parameter(property = "siteSpeed.url", required = true)
53      private URL url;
54  
55      /** The result base directory. */
56      @Parameter(property = "siteSpeed.outputDir", required = false, defaultValue = "target/sitespeed-result")
57      private File outputDir;
58  
59      /**
60       * Main Mojo method.
61       * 
62       * @throws MojoExecutionException
63       *             in case of execution error
64       * @throws MojoFailureException
65       *             in case of execution failure
66       */
67      @Override
68      public final void execute() throws MojoExecutionException,
69              MojoFailureException {
70          getLog().info("siteSpeedPath=[" + siteSpeedPath + "]");
71          getLog().info("phantomJSPath=[" + phantomJSPath + "]");
72          getLog().info("url=[" + url.toExternalForm() + "]");
73          getLog().info("outputDir=[" + outputDir + "]");
74          BufferedReader reader = null;
75          try {
76              ProcessBuilder pb = new ProcessBuilder(constructCmdarray());
77              pb.directory(siteSpeedPath);
78              pb.environment().put("PATH",
79                      getPhantomJSBinDir() + ":" + pb.environment().get("PATH"));
80              pb.redirectErrorStream(true);
81              Process p = pb.start();
82              reader = new BufferedReader(new InputStreamReader(
83                      p.getInputStream()));
84              String line;
85              while ((line = reader.readLine()) != null) {
86                  getLog().info(line);
87              }
88              p.waitFor();
89          } catch (IOException ioe) {
90              throw new MojoExecutionException(
91                      "IOException when generating report", ioe);
92          } catch (InterruptedException ie) {
93              throw new MojoExecutionException(
94                      "InterruptedException when generating report", ie);
95          } finally {
96              Closeables.closeQuietly(reader);
97          }
98      }
99  
100     /**
101      * Construct the sitespeed.io command.
102      * 
103      * @return the command to run (including arguments)
104      */
105     private List<String> constructCmdarray() {
106         List<String> cmd = new ArrayList<String>();
107         cmd.add(getSiteSpeedBin().toString());
108         cmd.add("-u");
109         cmd.add(url.toExternalForm());
110         cmd.add("-r");
111         cmd.add(outputDir.toString());
112         return cmd;
113     }
114 
115     /**
116      * Get the <code>bin</code> directory of the PhantomJS distribution.
117      * 
118      * @return the PhantomJS <code>bin</code> directory
119      */
120     private File getPhantomJSBinDir() {
121         return new File(phantomJSPath, "bin");
122     }
123 
124     /**
125      * Get the <code>sitespeed.io</code> script.
126      * 
127      * @return the <code>sitespeed.io</code> script
128      */
129     private File getSiteSpeedBin() {
130         return new File(siteSpeedPath, "sitespeed.io");
131     }
132 }
